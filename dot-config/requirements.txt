# Arch Linux / Manjaro:
# sudo pacman -S hyprland hyprlock hyprpaper hypridle hyprpolkitagent \
#                waybar rofi-wayland wl-clipboard cliphist grimblast \
#                gammastep brightnessctl playerctl pavucontrol \
#                network-manager-applet blueman mako kitty nautilus \
#                obsidian slack fish tmux lazygit git picom \
#                ttf-sono ttf-jetbrains-mono-nerd gtk-engine-murrine
#
# flatpak install flathub app.zen_browser.zen
# flatpak install flathub com.slack.Slack

# Fedora:
# sudo dnf install waybar rofi-wayland wl-clipboard cliphist \
#                  gammastep brightnessctl playerctl pavucontrol \
#                  NetworkManager-applet blueman mako kitty nautilus \
#                  fish tmux lazygit git picom \
#                  jetbrains-mono-fonts
#
# hyprland is available via copr or from source

# Debian / Ubuntu:
# sudo apt install waybar rofi wl-clipboard cliphist \
